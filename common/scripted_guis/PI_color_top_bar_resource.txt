scripted_gui = {

	pi_color = {
		context_type = player_context
		
		window_name = "pi_color_container"
		
		visible = {
            always = yes
		}

		effects = {
            paint_icon_click = {
				#set_variable = { PIopen = 1 }
				if = {
					limit = { ROOT = { check_variable = { PIopen < 1.000 } } }
					ROOT = { set_variable = { PIopen = 1 } }
				}
				else = {
					ROOT = { set_variable = { PIopen = 0 } }
				}
			}
		}
	}
    
}
